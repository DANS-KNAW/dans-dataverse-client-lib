016package nl.knaw.dans.lib.dataverse;
017
018import org.apache.http.HttpResponse;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022import java.io.IOException;
023import java.nio.file.Path;
024import java.nio.file.Paths;
025
026public class SwordApi extends AbstractApi {
027
028    private static final Logger log = LoggerFactory.getLogger(SwordApi.class);
029
030    protected SwordApi(HttpClientWrapper httpClientWrapper) {
031        super(httpClientWrapper.sendApiTokenViaBasicAuth());
032        log.trace("ENTER");
033    }
034
035    /**
036     * Deletes a file from the current draft of the dataset.
037     *
038     * [Dataverse API Guide]: https://guides.dataverse.org/en/latest/api/sword.html#delete-a-file-by-database-id
039     *
040     * @param databaseId the database ID of the file to delete.
041     *                   To look up use DatasetApi#listFiles.
042     * @return
043     */
044    public HttpResponse deleteFile(int databaseId) throws IOException, DataverseException {
045        log.trace("ENTER");
046        Path path = Paths.get("/dvn/api/data-deposit/v1.1/swordv2/edit-media/file/" + databaseId);
047        return httpClientWrapper.delete(path);
048    }
049}

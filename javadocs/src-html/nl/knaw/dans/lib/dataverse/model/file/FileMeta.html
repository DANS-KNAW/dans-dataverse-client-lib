016package nl.knaw.dans.lib.dataverse.model.file;
017
018import nl.knaw.dans.lib.dataverse.model.file.prestaged.Checksum;
019import nl.knaw.dans.lib.dataverse.model.file.prestaged.PrestagedFile;
020
021import java.util.List;
022
023public class FileMeta {
024
025    private String label;
026    private String description;
027    private String directoryLabel;
028    private int version;
029    private int datasetVersionId;
030    private Boolean restricted;
031    private List<String> categories; // TODO Enum or vocab? https://guides.dataverse.org/en/latest/user/dataset-management.html?highlight=category#file-tags
032    private DataFile dataFile;
033    private Boolean forceReplace;
034
035    public FileMeta() {
036    }
037
038    public PrestagedFile toPrestagedFile() {
039        if (dataFile == null)
040            throw new IllegalArgumentException("Cannot convert to PrestagedFile if dataFile is null");
041        PrestagedFile p = new PrestagedFile();
042        p.setStorageIdentifier(dataFile.getStorageIdentifier());
043        p.setFileName(label);
044        p.setMimeType(dataFile.getContentType());
045        p.setChecksum(new Checksum(dataFile.getChecksum().getType(), dataFile.getChecksum().getValue()));
046        p.setDescription(description);
047        p.setDirectoryLabel(directoryLabel);
048        p.setCategories(categories);
049        p.setRestrict(restricted);
050        p.setForceReplace(forceReplace);
051        return p;
052    }
053
054    public String getLabel() {
055        return label;
056    }
057
058    public void setLabel(String label) {
059        this.label = label;
060    }
061
062    public String getDescription() {
063        return description;
064    }
065
066    public void setDescription(String description) {
067        this.description = description;
068    }
069
070    public String getDirectoryLabel() {
071        return directoryLabel;
072    }
073
074    public void setDirectoryLabel(String directoryLabel) {
075        this.directoryLabel = directoryLabel;
076    }
077
078    public Boolean getRestricted() {
079        return restricted;
080    }
081
082    public void setRestricted(Boolean restricted) {
083        this.restricted = restricted;
084    }
085
086    public List<String> getCategories() {
087        return categories;
088    }
089
090    public void setCategories(List<String> categories) {
091        this.categories = categories;
092    }
093
094    public Boolean getForceReplace() {
095        return forceReplace;
096    }
097
098    public void setForceReplace(Boolean forceReplace) {
099        this.forceReplace = forceReplace;
100    }
101
102    public DataFile getDataFile() {
103        return dataFile;
104    }
105
106    public void setDataFile(DataFile dataFile) {
107        this.dataFile = dataFile;
108    }
109
110    public int getDatasetVersionId() {
111        return datasetVersionId;
112    }
113
114    public void setDatasetVersionId(int datasetVersionId) {
115        this.datasetVersionId = datasetVersionId;
116    }
117
118    public int getVersion() {
119        return version;
120    }
121
122    public void setVersion(int version) {
123        this.version = version;
124    }
125
126}

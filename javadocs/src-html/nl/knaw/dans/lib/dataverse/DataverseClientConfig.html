016package nl.knaw.dans.lib.dataverse;
017
018import java.net.URI;
019
020public class DataverseClientConfig {
021    private final URI baseUrl;
022    private final String apiToken;
023    private final int awaitLockStateMaxNumberOfRetries;
024    private final int awaitLockStateMillisecondsBetweenRetries;
025    private final String unblockKey;
026
027    /**
028     * Configuration settings for the {@link DataverseClient}.
029     *
030     * @param baseUrl                                  the base URL of the Dataverse server to communicate with
031     * @param apiToken                                 the API token used for authorization
032     * @param awaitLockStateMaxNumberOfRetries         the maximum number of tries for {@link DatasetApi#awaitLock(String)} API (default 30)
033     * @param awaitLockStateMillisecondsBetweenRetries the number or milliseconds to wait between tries for {@link DatasetApi#awaitLock(String)} API (default 500)
034     * @param unblockKey                               a key required for admin tasks when not running on localhost
035     */
036    public DataverseClientConfig(URI baseUrl, String apiToken, int awaitLockStateMaxNumberOfRetries, int awaitLockStateMillisecondsBetweenRetries, String unblockKey) {
037        this.baseUrl = baseUrl;
038        this.apiToken = apiToken;
039        this.awaitLockStateMaxNumberOfRetries = awaitLockStateMaxNumberOfRetries;
040        this.awaitLockStateMillisecondsBetweenRetries = awaitLockStateMillisecondsBetweenRetries;
041        this.unblockKey = unblockKey;
042    }
043
044    /**
045     * Configuration settings for the {@link DataverseClient}.
046     *
047     * @param baseUrl    the base URL of the Dataverse server to communicate with
048     * @param apiToken   the API token used for authorization
049     * @param unblockKey a key required for admin tasks when not running on localhost
050     */
051    public DataverseClientConfig(URI baseUrl, String apiToken, String unblockKey) {
052        this(baseUrl, apiToken, 30, 500, unblockKey);
053    }
054
055    /**
056     * Configuration settings for the {@link DataverseClient}.
057     *
058     * @param baseUrl  the base URL of the Dataverse server to communicate with
059     * @param apiToken the API token used for authorization
060     */
061    public DataverseClientConfig(URI baseUrl, String apiToken) {
062        this(baseUrl, apiToken, 30, 500, null);
063    }
064
065    /**
066     * Configuration settings for the {@link DataverseClient}. No API token is specified, so the client will only able to access endpoints that require no account.
067     *
068     * @param baseUrl the base URL of the Dataverse server to communicate with
069     */
070    public DataverseClientConfig(URI baseUrl) {
071        this(baseUrl, null);
072    }
073
074    public URI getBaseUrl() {
075        return baseUrl;
076    }
077
078    public String getApiToken() {
079        return apiToken;
080    }
081
082    int getAwaitLockStateMaxNumberOfRetries() {
083        return awaitLockStateMaxNumberOfRetries;
084    }
085
086    int getAwaitLockStateMillisecondsBetweenRetries() {
087        return awaitLockStateMillisecondsBetweenRetries;
088    }
089
090    public String getUnblockKey() {
091        return unblockKey;
092    }
093}

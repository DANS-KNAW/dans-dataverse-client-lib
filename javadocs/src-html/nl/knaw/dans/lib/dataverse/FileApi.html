053    /**
054     * [Dataverse API Guide]: https://guides.dataverse.org/en/latest/api/native-api.html#replacing-files
055     *
056     * @param optDataFile
057     * @param optFileMetadata json
058     * @return
059     * @throws IOException        when I/O problems occur during the interaction with Dataverse
060     * @throws DataverseException when Dataverse fails to perform the request
061     */
062    public DataverseHttpResponse<FileList> replaceFileItem(Optional<File> optDataFile, Optional<String> optFileMetadata) throws IOException, DataverseException {
063        logger.trace("{} {}", optDataFile, optFileMetadata);
064        if (!optDataFile.isPresent() && !optFileMetadata.isPresent())
065            throw new IllegalArgumentException("At least one of file data and file metadata must be provided.");
066        MultipartEntityBuilder builder = MultipartEntityBuilder.create();
067        optDataFile.ifPresent(f -> builder.addPart("file", new FileBody(f, ContentType.APPLICATION_OCTET_STREAM, f.getName())));
068        optFileMetadata.ifPresent(m -> builder.addPart("jsonData", new InputStreamBody(new ByteArrayInputStream(m.getBytes()), ContentType.APPLICATION_JSON, "jsonData")));
069        return httpClientWrapper.post(subPath("replace"), builder.build(), (emptyMap()), new HashMap<>(), FileList.class);
070    }
071
072    /**
073     * [Dataverse API Guide]: https://guides.dataverse.org/en/latest/api/native-api.html#updating-file-metadata
074     *
075     * @param json the file metadata
076     * @return
077     * @throws IOException        when I/O problems occur during the interaction with Dataverse
078     * @throws DataverseException when Dataverse fails to perform the request
079     */
080    public DataverseHttpResponse<HashMap> updateMetadata(String json) throws IOException, DataverseException {
081        MultipartEntityBuilder builder = MultipartEntityBuilder.create();
082        builder.setMode(HttpMultipartMode.BROWSER_COMPATIBLE);
083        builder.addBinaryBody("jsonData", json.getBytes(StandardCharsets.UTF_8), ContentType.APPLICATION_JSON, "jsonData");
084        return httpClientWrapper.post(subPath("metadata"), builder.build(), params(emptyMap()), new HashMap<>(), HashMap.class);
085    }

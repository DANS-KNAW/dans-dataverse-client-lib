016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.databind.ObjectMapper;
019import com.fasterxml.jackson.databind.module.SimpleModule;
020import nl.knaw.dans.lib.dataverse.model.dataset.MetadataField;
021import nl.knaw.dans.lib.dataverse.model.dataverse.DataverseItem;
022import nl.knaw.dans.lib.dataverse.model.search.ResultItem;
023import org.apache.http.client.HttpClient;
024import org.apache.http.impl.client.HttpClients;
025
026/**
027 * Object that lets your code talk to a Dataverse server.
028 */
029public class DataverseClient {
030
031    private final HttpClientWrapper httpClientWrapper;
032    private SearchApi searchApi;
033
034    /**
035     * Creates a DataverseClient.
036     *
037     * @param config configuration for this DataverseClient
038     */
039    public DataverseClient(DataverseClientConfig config) {
040        this(config, null, null);
041    }
042
043    /**
044     * Creates a DataverseClient with a custom HttpClient.
045     *
046     * @param config       configuration for this DataverseClient
047     * @param httpClient   the `org.apache.http.client.HttpClient` to use when interacting with Dataverse, or null to use a default HttpClient
048     * @param objectMapper the Jackson object mapper to use, or null to use a default mapper
049     */
050    public DataverseClient(DataverseClientConfig config, HttpClient httpClient, ObjectMapper objectMapper) {
051        ObjectMapper mapper = objectMapper == null ? new ObjectMapper() : objectMapper;
052        SimpleModule module = new SimpleModule();
053        // TODO: How to get rid of type warnings?
054        // TODO: Create proper Jackson module for this?
055        // TODO: Make use of this deserializer optional through system property?
056        module.addDeserializer(MetadataField.class, new MetadataFieldDeserializer());
057        module.addDeserializer(DataverseItem.class, new DataverseItemDeserializer());
058        module.addDeserializer(ResultItem.class, new ResultItemDeserializer(mapper));
059        mapper.registerModule(module);
060        this.httpClientWrapper = new HttpClientWrapper(config, httpClient == null ? HttpClients.createDefault() : httpClient, mapper);
061    }
062
063    public WorkflowsApi workflows() {
064        return new WorkflowsApi(httpClientWrapper);
065    }
066
067    public DatasetApi dataset(String pid) {
068        return dataset(pid, null);
069    }
070
071    public DatasetApi dataset(String pid, String invocationId) {
072        return new DatasetApi(httpClientWrapper, pid, true, invocationId);
073    }
074
075    public DatasetApi dataset(int pid) {
076        return dataset(pid, null);
077    }
078
079    public DatasetApi dataset(int pid, String invocationId) {
080        return new DatasetApi(httpClientWrapper, Integer.toString(pid), false, invocationId);
081    }
082
083    public DataverseApi dataverse(String alias) {
084        return new DataverseApi(httpClientWrapper, alias);
085    }
086
087    public AdminApi admin() {
088        return new AdminApi(httpClientWrapper);
089    }
090
091    public SwordApi sword() {
092        return new SwordApi(httpClientWrapper);
093    }
094
095    public FileApi file(int id) {
096        return new FileApi(httpClientWrapper, String.valueOf(id), false);
097    }
098
099    public DataAccessRequestsApi accessRequests(String pid) {
100        return new DataAccessRequestsApi(httpClientWrapper, pid, true);
101    }
102
103    public DataAccessRequestsApi accessRequests(int id) {
104        return new DataAccessRequestsApi(httpClientWrapper, String.valueOf(id), false);
105    }
106
107    public BasicFileAccessApi basicFileAccess(String pid) {
108        return new BasicFileAccessApi(httpClientWrapper, pid, true);
109    }
110
111    public BasicFileAccessApi basicFileAccess(int id) {
112        return new BasicFileAccessApi(httpClientWrapper, Integer.toString(id), false);
113    }
114
115    public SearchApi search() {
116        if (searchApi == null)
117            searchApi = new SearchApi(httpClientWrapper);
118        return searchApi;
119    }
120}

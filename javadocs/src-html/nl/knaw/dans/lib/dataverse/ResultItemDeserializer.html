016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.core.JsonParser;
019import com.fasterxml.jackson.core.JsonProcessingException;
020import com.fasterxml.jackson.databind.DeserializationContext;
021import com.fasterxml.jackson.databind.JavaType;
022import com.fasterxml.jackson.databind.JsonNode;
023import com.fasterxml.jackson.databind.ObjectMapper;
024import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
025import nl.knaw.dans.lib.dataverse.model.file.Checksum;
026import nl.knaw.dans.lib.dataverse.model.search.Contact;
027import nl.knaw.dans.lib.dataverse.model.search.DatasetResultItem;
028import nl.knaw.dans.lib.dataverse.model.search.DataverseResultItem;
029import nl.knaw.dans.lib.dataverse.model.search.FileResultItem;
030import nl.knaw.dans.lib.dataverse.model.search.ResultItem;
031
032import java.io.IOException;
033import java.net.URI;
034import java.util.List;
035import java.util.Map;
036
037public class ResultItemDeserializer extends StdDeserializer {
038    private ObjectMapper mapper;
039
040    public ResultItemDeserializer(ObjectMapper mapper) {
041        this(null, mapper);
042    }
043
044    protected ResultItemDeserializer(Class<?> vc, ObjectMapper mapper) {
045
046        super(vc);
047        this.mapper = mapper;
048    }
049
050    @Override
051    public Object deserialize(JsonParser p, DeserializationContext ctxt) throws IOException {
052        JsonNode node = p.getCodec().readTree(p);
053        String type = node.get("type").asText();
054
055        if ("dataverse".equalsIgnoreCase(type)) {
056            DataverseResultItem resultItem = new DataverseResultItem();
057            setCommonFields(node, resultItem);
058            resultItem.setIdentifier(getStringValue(node, "identifier"));
059            return resultItem;
060        }
061        else if ("dataset".equalsIgnoreCase(type)) {
062            DatasetResultItem resultItem = new DatasetResultItem();
063            setCommonFields(node, resultItem);
064            resultItem.setGlobalId(getStringValue(node, "global_id"));
065            resultItem.setPublisher(getStringValue(node, "publisher"));
066            resultItem.setCitationHtml(getStringValue(node, "citationHtml"));
067            resultItem.setIdentifierOfDataverse(getStringValue(node, "identifier_of_dataverse"));
068            resultItem.setNameOfDataverse(getStringValue(node, "name_of_dataverse"));
069            resultItem.setCitation(getStringValue(node, "citation"));
070            resultItem.setStorageIdentifier(getStringValue(node, "storageIdentifier"));
071            resultItem.setSubjects(getListValue(node, "subjects", String.class));
072            resultItem.setFileCount(getIntegerValue(node, "fileCount"));
073            resultItem.setVersionId(getIntegerValue(node, "versionId"));
074            resultItem.setVersionState(getStringValue(node, "versionState"));
075            resultItem.setMajorVersion(getIntegerValue(node, "majorVersion"));
076            resultItem.setMinorVersion(getIntegerValue(node, "minorVersion"));
077            resultItem.setCreatedAt(getStringValue(node, "created_at"));
078            resultItem.setUpdatedAt(getStringValue(node, "updated_at"));
079            resultItem.setContacts(getListValue(node, "contacts", Contact.class));
080            resultItem.setAuthors(getListValue(node, "authors", String.class));
081            resultItem.setPublications(getListOfObjects(node, "publications"));
082            return resultItem;
083        }
084        else if ("file".equalsIgnoreCase(type)) {
085            FileResultItem resultItem = new FileResultItem();
086            setCommonFields(node, resultItem);
087            resultItem.setFileType(getStringValue(node, "file_type"));
088            resultItem.setFileContentType(getStringValue(node, "file_content_type"));
089            resultItem.setSizeInBytes(mapper.treeToValue(node.get("size_in_bytes"), Long.class));
090            resultItem.setChecksum(mapper.treeToValue(node.get("checksum"), Checksum.class));
091            resultItem.setDatasetName(getStringValue(node, "dataset_name"));
092            resultItem.setDatasetId(getStringValue(node, "dataset_id"));
093            resultItem.setDatasetPersistentId(getStringValue(node, "dataset_persistent_id"));
094            resultItem.setDatasetCitation(getStringValue(node, "dataset_citation"));
095            return resultItem;
096        }
097        else {
098            throw new IllegalArgumentException("Cannot this type of ResultItem: " + type);
099        }
100    }
101
102    private void setCommonFields(JsonNode node, ResultItem resultItem) throws JsonProcessingException {
103        resultItem.setName(getStringValue(node, "name"));
104        resultItem.setUrl(getUriValue(node, "url"));
105        resultItem.setDescription(getStringValue(node, "description"));
106        resultItem.setPublishedAt(getStringValue(node, "published_at"));
107    }
108
109    private String getStringValue(JsonNode node, String name) throws JsonProcessingException {
110        return mapper.treeToValue(node.get(name), String.class);
111    }
112
113    private URI getUriValue(JsonNode node, String name) throws JsonProcessingException {
114        return mapper.treeToValue(node.get(name), URI.class);
115    }
116
117    private Integer getIntegerValue(JsonNode node, String name) throws JsonProcessingException {
118        return mapper.treeToValue(node.get(name), Integer.class);
119    }
120
121    private <E> List<E> getListValue(JsonNode node, String name, Class<E> containedClass) throws JsonProcessingException {
122        JavaType t = mapper.getTypeFactory().constructParametricType(List.class, containedClass);
123        // writing the node to String and and then parsing it back again seems to be the only way for parametric types.
124        if (node.get(name) == null)
125            return null;
126        else
127            return mapper.readValue(node.get(name).toString(), t);
128    }
129
130    // Generic getter for "objects", currently only used for the mysterious field "publications"
131    private List<Map<Object, Object>> getListOfObjects(JsonNode node, String name) throws JsonProcessingException {
132        JavaType t = mapper.getTypeFactory().constructParametricType(List.class, Map.class);
133        // writing the node to String and and then parsing it back again seems to be the only way for parametric types.
134        if (node.get(name) == null)
135            return null;
136        else
137            return mapper.readValue(node.get(name).toString(), t);
138    }
